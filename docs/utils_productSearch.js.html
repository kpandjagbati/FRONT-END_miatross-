function normalize(str) {
  return String(str)
    .toLowerCase()
    .normalize('NFD')
    .replace(/[\u0300-\u036f]/g, '')
    .trim()
}

function tokenize(str) {
  return normalize(str).split(/[\s&\-–—/]+/).filter(Boolean)
}

/** Construit la liste de termes à chercher à partir de la requête utilisateur */
export function buildSearchTerms(query) {
  const q = normalize(query)
  if (!q) return []

  const terms = new Set([q, ...tokenize(q)])

  for (const [key, aliases] of Object.entries(SEARCH_EXPANSIONS)) {
    const normKey = normalize(key)
    const keyMatches = normKey.includes(q) || q.includes(normKey) ||
      tokenize(q).some(t => normKey.includes(t) || t.includes(normKey))
    const aliasMatches = aliases.some(alias => {
      const normAlias = normalize(alias)
      return normAlias.includes(q) || q.includes(normAlias) ||
        tokenize(q).some(t => normAlias.includes(t) || t.includes(normAlias))
    })

    if (keyMatches || aliasMatches) {
      terms.add(normKey)
      aliases.forEach(alias => terms.add(normalize(alias)))
    }
  }

  for (const [catKey, keywords] of Object.entries(CATEGORY_KEYWORDS)) {
    const label = normalize(CATEGORY_LABELS[catKey])
    const catMatch = keywords.some(kw => {
      const n = normalize(kw)
      return n.includes(q) || q.includes(n) || tokenize(q).some(t => n.includes(t) || t.includes(n))
    }) || label.includes(q) || q.includes(label) || catKey.includes(q)

    if (catMatch) {
      terms.add(catKey)
      terms.add(label)
      keywords.forEach(kw => terms.add(normalize(kw)))
    }
  }

  return [...terms]
}

/** Vérifie si un produit correspond à la recherche */
export function productMatchesSearch(product, query) {
  const q = normalize(query)
  if (!q) return true

  const searchTerms = buildSearchTerms(query)
  const categoryKey = product.categoryKey || ''
  const categoryLabel = product.category || CATEGORY_LABELS[categoryKey] || ''

  const haystack = normalize([
    product.name,
    categoryLabel,
    categoryKey,
    ...(CATEGORY_KEYWORDS[categoryKey] || []),
    ...tokenize(product.name),
  ].join(' '))

  return searchTerms.some(term => {
    if (term.length < 2) return haystack.includes(term)
    return haystack.includes(term) || term.split(' ').some(part => haystack.includes(part))
  })
}
